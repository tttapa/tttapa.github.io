135 Drone::Controller::clampAttitude(const Attitude::LQRController::VecU_t &u_raw,
136  const Altitude::LQRController::VecU_t &u_alt) {
137  ColVector<3> u = u_raw;
138  double other_max = 1 - abs(u_alt);
139  double other_actual = abs(u_raw[0]) + abs(u_raw[1]) + abs(u_raw[2]);
140  if (other_actual > other_max) {
141  u[0] = {other_max / other_actual * u_raw[0]};
142  u[1] = {other_max / other_actual * u_raw[1]};
143  u[2] = {other_max / other_actual * u_raw[2]};
144  }
145  other_actual = abs(u[0]) + abs(u[1]) + abs(u[2]);
146  double e = numeric_limits<double>::epsilon() + 1.0;
147  if (other_actual > abs(u_alt)) {
148  u[0] = {double(u_alt) / other_actual * u[0] / e};
149  u[1] = {double(u_alt) / other_actual * u[1] / e};
150  u[2] = {double(u_alt) / other_actual * u[2] / e};
151  }
152  return u;
153 }
