140  // If the range contains no discontinuities
141  // ├───┬───┬───╤═══╤═══╤═══╪═══╤═══╤═══╤───┬───┬───┤
142  // 0 C-N C C+N R-1
143  // C = center
144  // N = number of steps around the center
145  // R = resolution (R steps = 720°)
146  if (centerAngleIndex - N < angle_t::resolution() &&
147  centerAngleIndex + N < angle_t::resolution()) {
148  uint initialAngleIndex = centerAngleIndex - N;
149  for (uint i = 0; i <= 2 * N; ++i)
150  houghRes.at(i) = hough(px, angle_t(initialAngleIndex + i));
151  }
152  // If the 0-2π discontinuity is in the first part of the range
153  // (< centerAngle)
154  // ╞═══╪═══╤═══╤═══╤───┬───┬───┬───┬───┬───┬───╤═══╡

157  uint initialAngleIndex =
158  angle_t::resolution() + centerAngleIndex - N;
159  uint i = 0;
160  for (uint j = initialAngleIndex; j < angle_t::resolution();

163  for (uint j = 0; j <= centerAngleIndex + N; ++i, ++j)
164  houghRes.at(i) = hough(px, angle_t(j));

166  // If the 0-2π discontinuity is in the second part of the range
167  // (> centerAngle)

316  uint maxLineGap = MAX_GAP) const {

328  // L2 L1

332  maxLineGap / 2, false);

334  maxLineGap / 2, false);
335  uint widthLower2 = getWidthAtPointOnLine(pointOnLine, oppositeAngle, //
336  maxLineGap / 2, true);
337

339  uint widthUpper = std::max(widthUpper1, widthUpper2);
340  uint widthLower = std::max(widthLower1, widthLower2);
341

343  // invalid: it is probably a large white spot, not a thin line
344  if (widthUpper >= MAX_LINE_WIDTH || widthLower >= MAX_LINE_WIDTH)
345  return std::nullopt;
346 
347  // Use the difference between the two widths to move the given point
348  // towards the center
349  int middlePointCorrection_x2 = widthUpper - widthLower;
350  uint middlePointCorrDistance = std::abs(middlePointCorrection_x2) / 2;
351  // Should we move up or down? (+90° or -90° from the estimated angle)
352  bool corrDirection = middlePointCorrection_x2 > 0;
353  CosSin corrAngle = lineAngle.perpendicular(corrDirection);
