  55 // If it's a Teensy board
  56 #if defined(CORE_TEENSY)
  57     MIDI_read_t read() override {
  58         while (1) {
  59             if (rx_packet == nullptr) { // If there's no previous packet
  60                 if (!usb_configuration) // Check USB configuration
  61                     return NO_MESSAGE;
  62                 rx_packet = usb_rx(
  63                     MIDI_RX_ENDPOINT); // Read a new packet from the USB buffer
  64                 if (rx_packet == nullptr) { // If there's no new packet, return
  65                     return NO_MESSAGE;
  66                 }
  67                 if (rx_packet->len < 4) {
  68                     // If the lenght is less than 4, it's not a valid MIDI USB
  69                     // packet
  70                     usb_free(rx_packet); // Free the packet
  71                     rx_packet = nullptr; // Read new packet next time around
  72                     return NO_MESSAGE;
  73                 }
  74             }
  75 
  76             size_t index = rx_packet->index;
  77             uint8_t *data = rx_packet->buf + index; // A pointer to this packet
  78 
  79             MIDI_read_t parseResult = parser.parse(data);
  80 
  81             index += 4;
  82             if (index < rx_packet->len) {
  83                 // If the packet is longer than 4 bytes
  84                 rx_packet->index = index; // Change the index and read the next
  85                                           // four bytes on next read
  86             } else {
  87                 // If the packet was only 4 bytes long
  88                 usb_free(rx_packet);                  // Free the packet
  89                 rx_packet = usb_rx(MIDI_RX_ENDPOINT); // Read the next packet
  90             }
  91             if (parseResult != NO_MESSAGE)
  92                 return parseResult;
  93         }
  94     }
