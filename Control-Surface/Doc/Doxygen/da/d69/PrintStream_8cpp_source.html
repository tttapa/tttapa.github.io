209 template <class T>
210 void printHex(Print &printer, T val)
211 {
212  if (showbasePrintStream)
213  printer.print("0x");
214  bool nonZero = false;
215  for (int i = sizeof(val) - 1; i >= 0; i--)
216  {
217  uint8_t currByte = ((uint8_t *)&val)[i];
218  if (currByte != 0 || i == 0)
219  nonZero = true;
220  if (leadingZerosPrintStream || nonZero) {
221  printer.print(nibble_to_hex(currByte >> 4));
222  printer.print(nibble_to_hex(currByte));
223  if (byteSeparatorPrintStream && i)
224  printer.print(byteSeparatorPrintStream);
225  }
226  }
227 }

229 template <class T>
230 void printBin(Print &printer, T val)
231 {
232  if (showbasePrintStream)
233  printer.print("0b");
234  bool nonZero = false;
235  for (int i = sizeof(val) - 1; i >= 0; i--)
236  {
237  uint8_t currByte = ((uint8_t *)&val)[i];
238  for (int j = 7; j >= 0; j--)
239  {
240  uint8_t currBit = currByte & 0x80;
241  if (currBit != 0 || (i == 0 && j == 0))
242  nonZero = true;
243  if (leadingZerosPrintStream || nonZero)
244  printer.print(currBit ? '1' : '0');
245  currByte <<= 1;
246  }
247  if (byteSeparatorPrintStream && i && (leadingZerosPrintStream || nonZero))
248  printer.print(byteSeparatorPrintStream);
249  }
